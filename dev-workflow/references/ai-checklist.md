## 1. 新模块开发前检查
### 1.1 立即检查（必做）
```python
def 新模块开发前检查(模块名, 功能描述):
    """创建新模块前必须检查的项目"""

    检查结果 = {}

    # 检查 1: 是否需要新数据结构？
    需要的数据结构 = 分析输入输出(功能描述)
    for ds in 需要的数据结构:
            if not exists_in_data_structure_spec(ds):
                检查结果["缺失数据结构"] = 检查结果.get("缺失数据结构", []) + [ds]

    # 检查 2: 是否需要新接口？
    需要的接口 = 分析接口需求(功能描述)
    for api in 需要的接口:
            if not exists_in_interface_spec(api):
                检查结果["缺失接口"] = 检查结果.get("缺失接口", []) + [api]

    # 检查 3: 模块定位是否清晰？
    if not 模块定位清晰(功能描述):
                检查结果["需要明确"] = 检查结果.get("需要明确", []) + ["模块定位"]

    # 输出检查结果
    if 检查结果:
        print("⚠️  新模块开发前需要先完成：")
        for 类型, 项目列表 in 检查结果.items():
            print(f"  {类型}: {', '.join(项目列表)}")
        return False
    else:
        print("✅ 新模块开发前置条件已满足")
        return True
```

### 1.2 检查清单
- [ ] 是否需要新的数据结构？→ 是否已在数据结构规范中定义？
- [ ] 是否需要新的接口？→ 是否已在接口规范中定义？
- [ ] 模块定位是否清晰？→ 是否明确了"做什么/不做什么"？
- [ ] 是否了解了依赖关系？→ 上游依赖？下游模块？
- [ ] 文档模板是否准备好？→ 是否知道使用哪个模板？

## 2. 修改代码前检查
### 2.1 立即检查（必做）
```python
def 修改代码前检查(文件路径, 变更内容):
    """修改代码前必须检查的项目"""

    检查结果 = {}

    # 检查 1: 分析变更类型
    变更类型 = 分析变更类型(变更内容)
    检查结果["变更类型"] = 变更类型

    # 检查 2: 是否涉及数据结构变更？
    if 变更类型 == "数据结构变更":
        if not exists("doc/02-技术规范/数据结构规范.md"):
            检查结果["缺失文档"] = ["数据结构规范"]
        else:
            检查结果["需要更新"] = 检查结果.get("需要更新", []) + ["数据结构规范"]

    # 检查 3: 是否涉及接口变更？
    if 变更类型 == "接口变更":
        if not exists("doc/02-技术规范/接口规范.md"):
            检查结果["缺失文档"] = 检查结果.get("缺失文档", []) + ["接口规范"]
        else:
            检查结果["需要更新"] = 检查结果.get("需要更新", []) + ["接口规范"]

    # 检查 4: 影响分析
    if 变更类型 in ["数据结构变更", "接口变更"]:
        受影响的文档 = find_all_references(变更内容)
        if 受影响的文档:
            检查结果["受影响文档"] = 受影响的文档

    # 输出检查结果
    if 检查结果.get("缺失文档"):
        print("❌ 无法执行：缺少必要文档")
        return False
    elif 检查结果.get("需要更新") or 检查结果.get("受影响文档"):
        print("⚠️  检测到变更，需要先更新文档：")
        for 类型 in ["需要更新", "受影响文档"]:
            if 检查结果.get(类型):
                print(f"  {类型}: {', '.join(检查结果[类型])}")
        return "需要更新文档"
    else:
        print("✅ 可以直接修改代码")
        return True
```

### 2.2 检查清单
- [ ] 变更类型是什么？→ 数据结构/接口/算法/参数/简单修复？
- [ ] 是否涉及数据结构？→ 是否需要更新数据结构规范？
- [ ] 是否涉及接口？→ 是否需要更新接口规范？
- [ ] 影响范围多大？→ 哪些文档/代码会受影响？
- [ ] 是否需要影响分析？→ 列出所有受影响的文档

## 3. 创建文档前检查
### 3.1 立即检查（必做）
```python
def 创建文档前检查(文档类型, 文档名称):
    """创建文档前必须检查的项目"""

    检查结果 = {}

    # 检查 1: 文档是否已存在？
    if exists(f"doc/{文档类型}/{文档名称}.md"):
        检查结果["已存在"] = True
        print("❌ 文档已存在，无需创建")
        return False

    # 检查 2: 文档层级是否正确？
    文档层级 = 确定文档层级(文档类型)
    if not 文档层级:
        检查结果["错误"] = "无法确定文档层级"
        print("❌ 无法确定文档层级")
        return False

    # 检查 3: 是否有对应模板？
    模板 = 查找模板(文档类型)
    if not 模板:
        检查结果["警告"] = "未找到模板，将使用默认结构"
    else:
        检查结果["模板"] = 模板

    # 检查 4: 上位文档是否存在？
    上位文档 = 确定上位文档(文档层级)
    if 上位文档 and not exists(上位文档):
        检查结果["缺失上位文档"] = 上位文档

    # 输出检查结果
    if 检查结果.get("缺失上位文档"):
        print(f"⚠️  上位文档不存在：{上位文档}")
        print("建议：先创建上位文档")
        return "需要上位文档"
    else:
        print("✅ 可以创建文档")
        return True
```

### 3.2 检查清单
- [ ] 文档是否已存在？→ 避免重复创建
- [ ] 文档层级是否明确？→ L1/L2/L3/L4/L5？
- [ ] 是否有对应模板？→ 使用正确的模板
- [ ] 上位文档是否存在？→ 系统设计/技术规范是否完备？
- [ ] 文档定位是否清晰？→ 知道回答什么问题？

## 4. Git 提交前检查
### 4.1 立即检查（必做）
```python
def Git提交前检查():
    """Git 提交前必须检查的项目"""

    检查结果 = {}

    # 检查 1: 代码与文档是否一致？
    变更的文件 = git_status()
    代码变更 = [f for f in 变更的文件 if f.startswith("src/")]
    文档变更 = [f for f in 变更的文件 if f.startswith("doc/")]

    if 代码变更 and not 文档变更:
        检查结果["不一致"] = "代码已修改但文档未更新"

    # 检查 2: 是否有临时文件？
    临时文件 = [f for f in 变更的文件 if f.startswith("temp/")]
    if 临时文件:
        检查结果["临时文件"] = 临时文件

    # 检查 3: 版本历史是否更新？
    if 文档变更:
        for doc in 文档变更:
            if not 文档包含版本历史更新(doc):
                检查结果["缺少版本历史"] = 检查结果.get("缺少版本历史", []) + [doc]

    # 检查 4: 提交信息是否规范？
    提交信息 = get_commit_message()
    if not 提交信息规范(提交信息):
        检查结果["提交信息不规范"] = True

    # 输出检查结果
    if 检查结果:
        print("⚠️  Git 提交前需要处理：")
        for 问题, 详情 in 检查结果.items():
            print(f"  {问题}: {详情}")
        return False
    else:
        print("✅ Git 提交检查通过")
        return True
```

### 4.2 检查清单
- [ ] 代码变更是否有对应的文档变更？
- [ ] 是否有临时文件被提交？
- [ ] 文档变更是否更新了版本历史？
- [ ] 提交信息是否符合规范？（feat/fix/docs/refactor/test/chore）
- [ ] 是否需要运行测试？

## 5. 文档完整性检查
### 5.1 立即检查（必做）
```python
def 文档完整性检查(文档路径):
    """检查文档是否完整"""

    文档内容 = read(文档路径)

    # 检查 1: 是否有文档定位？
    if not 文档内容.contains("文档定位"):
        print("❌ 缺少文档定位说明")
        return False

    # 检查 2: 是否有版本历史？
    if not 文档内容.contains("版本历史"):
        print("❌ 缺少版本历史")
        return False

    # 检查 3: 根据文档类型检查必需章节
    文档类型 = 确定文档类型(文档路径)
    必需章节 = {
        "模块设计": ["模块定位与边界", "输入输出数据规范", "核心算法", "验证方案"],
        "数据结构规范": ["数据结构定义", "字段说明", "约束条件"],
        "接口规范": ["接口签名", "参数说明", "返回值", "异常"]
    }

    for 章节 in 必需章节.get(文档类型, []):
        if not 文档内容.contains(章节):
            print(f"⚠️  缺少章节：{章节}")

    return True
```

### 5.2 检查清单
- [ ] 是否有文档定位说明？
- [ ] 是否有版本历史？
- [ ] 必需章节是否完整？（根据文档类型）
- [ ] 引用的链接是否有效？
- [ ] 是否有"待补充"标记？（应该补充完整或标记为草稿）

## 6. 一致性检查
### 6.1 立即检查（必做）
```python
def 一致性检查(触发原因="手动触发"):
    """执行文档一致性检查"""

    print(f"🔍 执行一致性检查（原因：{触发原因}）")
    问题列表 = []

    # 检查 1: 数据结构引用一致性
    for 模块文档 in glob("doc/03-模块设计/*.md"):
        引用的数据结构 = 提取数据结构引用(模块文档)
        for ds in 引用的数据结构:
            if not exists_in_data_structure_spec(ds):
                问题列表.append(f"❌ {模块文档} 引用的 {ds} 在数据结构规范中不存在")

    # 检查 2: 接口引用一致性
    for 模块文档 in glob("doc/03-模块设计/*.md"):
        引用的接口 = 提取接口引用(模块文档)
        for api in 引用的接口:
            if not exists_in_interface_spec(api):
                问题列表.append(f"❌ {模块文档} 引用的 {api} 在接口规范中不存在")

    # 检查 3: 临时文件位置
    for 文件 in glob("temp/**/*"):
        if 文件类型 in ["正式文档", "源代码"]:
            问题列表.append(f"⚠️  临时文件 {文件} 应该移到正式目录")

    # 输出结果
    if 问题列表:
        print("\n一致性问题：")
        for 问题 in 问题列表:
            print(f"  {问题}")
        return False
    else:
        print("✅ 一致性检查通过")
        return True
```

### 6.2 检查清单
- [ ] 所有模块文档引用的数据结构都在规范中定义？
- [ ] 所有模块文档引用的接口都在规范中定义？
- [ ] 临时文件都在 temp/ 目录？
- [ ] 没有重复定义（只引用不复制）？
- [ ] README 导航是否包含所有文档？

## 7. 快速决策表
| 场景 | 立即检查 | 通过条件 | 不通过处理 |
|------|---------|---------|-----------|
| 新模块开发 | 数据结构、接口、定位 | 全部已定义 | 先定义缺失项 |
| 修改代码 | 变更类型、影响范围 | 文档已更新或无需更新 | 先更新文档 |
| 创建文档 | 文档层级、模板、上位文档 | 上位文档存在 | 先创建上位文档 |
| Git 提交 | 代码文档一致性、临时文件 | 一致且无临时文件 | 更新文档、移除临时文件 |
| 文档完整性 | 定位、版本历史、必需章节 | 全部存在 | 补充缺失部分 |
| 一致性检查 | 数据结构、接口、临时文件 | 全部一致 | 修复不一致项 |

## 8. AI 执行前自检清单
在执行任何操作前，AI 应该问自己：

- [ ] **这是什么类型的任务？**→ 新模块/修改/Bug修复/POC/文档？
- [ ] **需要检查什么？**→ 根据任务类型选择检查项
- [ ] **是否有前置条件？**→ 数据结构/接口/上位文档是否完备？
- [ ] **是否涉及变更？**→ 需要影响分析吗？
- [ ] **文档是否完备？**→ 缺少什么章节？
- [ ] **是否符合规范？**→ 是否遵循了模板和格式要求？
- [ ] **是否需要用户确认？**→ 关键步骤是否询问了用户？
